    def comp PP_capillary_H_D as
        //import variables

        var t: second {pub: in};
        var H_global_L: dimensionless {pub: in};
        var H_global_R: dimensionless {pub: in};
        var H_L_out_RHS: dimensionless {pub: in};
        var H_R_out_LHS: dimensionless {pub: in};
        var gamma_mirror: dimensionless {pub: in};
        var tau_H_down: second {pub: in};
        var tau_H_mean: second {pub: in};

        //0.001

        var v_eps: m3_per_s {pub: in};
        var q_C_init: m3 {pub: in};

        //0

        var l: metre {pub: in};
        var r: metre {pub: in};
        var mu_plasma: Js_per_m3 {pub: in};
        var u_in: J_per_m3 {pub: in};
        var u_ext: J_per_m3 {pub: in};
        var u_out: J_per_m3 {pub: in};
        var tau_link: second {pub: in};
        var v_scale: m3_per_s {pub: in};
        var one_mm3: m3 {pub: in};
        var w_v: dimensionless {pub: out};
        var w_v_d: dimensionless {pub: out};

        w_v   = 0.5{dimensionless}+1{dimensionless}/pi*atan(v/v_scale);
        w_v_d = 0.5{dimensionless}+1{dimensionless}/pi*atan(v_d/v_scale);

        ///////////////////////////coupling with RHS vessel using ode to break 
        //instaneous coupling requirement/////////

        var H_link_R: dimensionless {init: H_global_L, pub: out};

        ode(H_link_R, t) = (H_L_out_RHS-H_link_R)/tau_link;

        //RHS means RHS vessel

        var H_link_L: dimensionless {init: H_global_L, pub: out};

        ode(H_link_L, t) = (H_R_out_LHS-H_link_L)/tau_link;

        //RHS means RHS vessel

        var H_L_in: dimensionless {pub: out};

        H_L_in = H_link_L;

        var H_R_in: dimensionless {pub: out};

        H_R_in = H_link_R;

        var H_up: dimensionless {pub: out};

        H_up = w_v_d*H_L_in+(1{dimensionless}-w_v_d)*H_R_in;

        var H_down: dimensionless {init: H_global_L, pub: out};
        var H_down_target: dimensionless {pub: out};

        H_down_target  = s_v_d*(H_mean+gamma_mirror*(H_mean-H_up))+(1{dimensionless}-s_v_d)*H_mean;
        ode(H_down, t) = (H_down_target-H_down)/tau_H_down;

        var s_v_d: dimensionless {pub: out};

        s_v_d = abs(v_d)/(abs(v_d)+v_eps);

        ///useful when flow =0, if 0, make H_down=H_mean

        var H_L_out: dimensionless {pub: out};
        var H_R_out: dimensionless {pub: out};

        H_L_out = (1{dimensionless}-w_v_d)*H_down+w_v_d*H_L_in;
        H_R_out = w_v_d*H_down+(1{dimensionless}-w_v_d)*H_R_in;

        var RBC_volume: m3 {init: RBC_volume_init, pub: out};

        ode(RBC_volume, t) = v*H_mass_L-v_d*H_mass_R;

        var RBC_volume_init: m3 {pub: out};

        RBC_volume_init = H_global_L*q_us;

        //pos/neg for RBC mass conservation only

        var v_pos: m3_per_s {pub: out};
        var v_neg: m3_per_s {pub: out};
        var v_d_pos: m3_per_s {pub: out};
        var v_d_neg: m3_per_s {pub: out};

        v_pos = w_v*v;

        // inflow from left (v>0)

        v_neg = (1{dimensionless}-w_v)*-v;

        // outflow to left (v<0)

        v_d_pos = w_v_d*v_d;

        // flow from left to right (v_d>0) => outflow on right

        v_d_neg = (1{dimensionless}-w_v_d)*-v_d;

        // inflow from right (v_d<0)

        //var H_target: dimensionless {pub: out};

        //H_target = RBC_volume/q;

        H_mean = RBC_volume/q;

        var H_mean: dimensionless {pub: out};

        //ode(H_mean, t) = (H_target-H_mean)/tau_H_mean;

        var H_mass_L: dimensionless {pub: out};
        var H_mass_R: dimensionless {pub: out};

        H_mass_L = w_v*H_L_in+(1{dimensionless}-w_v)*H_L_out;
        H_mass_R = w_v_d*H_R_out+(1{dimensionless}-w_v_d)*H_R_in;

        ////////////////

        var q_C: m3 {init: q_C_init, pub: out};
        var q: m3 {pub: out};
        var q_us: m3 {pub: out};

        q_us = pi*sqr(r)*l;
        q    = q_us+q_C;

        var C: m6_per_J {pub: out};

        C = pi*sqr(8.5e-9{metre})*l/133.322{J_per_m3};

        var Z: dimensionless {pub: out};

        Z = (0.8{dimensionless}+exp(-0.075{dimensionless}*2{dimensionless}*r*1e6{dimensionless}))*(-1{dimensionless}+1{dimensionless}/(1{dimensionless}+pow(10{dimensionless}, -11{dimensionless})*pow(2{dimensionless}*r*1e6{dimensionless}, 12{dimensionless})))+1{dimensionless}/(1{dimensionless}+pow(10{dimensionless}, -11{dimensionless})*pow(2{dimensionless}*r*1e6{dimensionless}, 12{dimensionless}));

        var mu_45: dimensionless;

        mu_45 = 6{dimensionless}*exp(-0.085{per_m}*2{dimensionless}*r*1e6{dimensionless})+3.2{dimensionless}-2.44{dimensionless}*exp(-0.06{dimensionless}*pow(2{per_m}*r*1e6{dimensionless}, 0.645{dimensionless}));

        var mu: Js_per_m3 {pub: out};

        mu = hem_dep_u_rel*mu_plasma;

        var hem_dep_u_rel: dimensionless;

        hem_dep_u_rel = 1{dimensionless}+(mu_45-1{dimensionless})*(pow(1{dimensionless}-H_mean, Z)-1{dimensionless})/(pow(1{dimensionless}-H_global_L, Z)-1{dimensionless})*sqr(2{dimensionless}*r*1e6{dimensionless}/(2{dimensionless}*r*1e6{dimensionless}-1.1{metre}));

        var R: Js_per_m6 {pub: out};

        R = 8{dimensionless}*mu*l/(pi*pow(r, 4{dimensionless}));

        var v: m3_per_s {pub: out};
        var v_mm3_s: mm3_per_s {pub: out};

        v_mm3_s = v/one_mm3;
        v       = (u_in-u)/(R/2{dimensionless});

        var u: J_per_m3 {pub: out};
        var u_mmHg: dimensionless {pub: out};

        u_mmHg      = u/133.322{J_per_m3};
        u           = q_C/C+u_ext;
        ode(q_C, t) = v-v_d;

        var v_d: m3_per_s {pub: out};
        var v_d_mm3_s: mm3_per_s {pub: out};

        v_d_mm3_s = v_d/one_mm3;
        v_d       = (u-u_out)/(R/2{dimensionless});
    enddef;

    def comp PP_pericyte_H_D as
        //import variables

        var tau_sig: second {pub: in};
        var t0: second {pub: in};
        var t: second {pub: in};
        var R_constriction_base: Js_per_m6 {pub: in};

        R_constriction = R_constriction_base+(R_constriction_final-R_constriction_base)/(1{dimensionless}+exp(-(t-t0)/tau_sig));

        var H_global_L: dimensionless {pub: in};

        H_L_out = (1{dimensionless}-w_v_d)*H_down+w_v_d*H_L_in;
        H_R_out = w_v_d*H_down+(1{dimensionless}-w_v_d)*H_R_in;

        var RBC_volume: m3 {init: RBC_volume_init, pub: out};

        ode(RBC_volume, t) = v*H_mass_L-v_d*H_mass_R;

        var RBC_volume_init: m3 {pub: out};

        RBC_volume_init = H_global_L*q_us;

        //pos/neg for RBC mass conservation only

        var v_pos: m3_per_s {pub: out};
        var v_neg: m3_per_s {pub: out};
        var v_d_pos: m3_per_s {pub: out};
        var v_d_neg: m3_per_s {pub: out};

        v_pos = w_v*v;

        // inflow from left (v>0)

        v_neg = (1{dimensionless}-w_v)*-v;

        // outflow to left (v<0)

        v_d_pos = w_v_d*v_d;

        // flow from left to right (v_d>0) => outflow on right

        v_d_neg = (1{dimensionless}-w_v_d)*-v_d;

        // inflow from right (v_d<0)

        //var H_target: dimensionless {pub: out};

        //H_target = RBC_volume/q;

        H_mean = RBC_volume/q;

        var H_mean: dimensionless {pub: out};

        //ode(H_mean, t) = (H_target-H_mean)/tau_H_mean;

        var H_mass_L: dimensionless {pub: out};
        var H_mass_R: dimensionless {pub: out};

        H_mass_L = w_v*H_L_in+(1{dimensionless}-w_v)*H_L_out;
        H_mass_R = w_v_d*H_R_out+(1{dimensionless}-w_v_d)*H_R_in;

        ////////////////

        var q_C: m3 {init: q_C_init, pub: out};
        var q: m3 {pub: out};
        var q_us: m3 {pub: out};

        q_us = pi*sqr(r)*l;
        q    = q_us+q_C;

        var C: m6_per_J {pub: out};

        C = pi*sqr(8.5e-9{metre})*l/133.322{J_per_m3};

        var Z: dimensionless {pub: out};

        Z = (0.8{dimensionless}+exp(-0.075{dimensionless}*2{dimensionless}*r*1e6{dimensionless}))*(-1{dimensionless}+1{dimensionless}/(1{dimensionless}+pow(10{dimensionless}, -11{dimensionless})*pow(2{dimensionless}*r*1e6{dimensionless}, 12{dimensionless})))+1{dimensionless}/(1{dimensionless}+pow(10{dimensionless}, -11{dimensionless})*pow(2{dimensionless}*r*1e6{dimensionless}, 12{dimensionless}));

        var mu_45: dimensionless;

        mu_45 = 6{dimensionless}*exp(-0.085{per_m}*2{dimensionless}*r*1e6{dimensionless})+3.2{dimensionless}-2.44{dimensionless}*exp(-0.06{dimensionless}*pow(2{per_m}*r*1e6{dimensionless}, 0.645{dimensionless}));

        var mu: Js_per_m3 {pub: out};

        mu = hem_dep_u_rel*mu_plasma;

        var hem_dep_u_rel: dimensionless;

        hem_dep_u_rel = 1{dimensionless}+(mu_45-1{dimensionless})*(pow(1{dimensionless}-H_mean, Z)-1{dimensionless})/(pow(1{dimensionless}-H_global_L, Z)-1{dimensionless})*sqr(2{dimensionless}*r*1e6{dimensionless}/(2{dimensionless}*r*1e6{dimensionless}-1.1{metre}));

        var R: Js_per_m6 {pub: out};

        R = 8{dimensionless}*mu*l/(pi*pow(r, 4{dimensionless}))+R_constriction;

        var v: m3_per_s {pub: out};
        var v_mm3_s: mm3_per_s {pub: out};

        v_mm3_s = v/one_mm3;
        v       = (u_in-u)/(R/2{dimensionless});

        var u: J_per_m3 {pub: out};
        var u_mmHg: dimensionless {pub: out};

        u_mmHg      = u/133.322{J_per_m3};
        u           = q_C/C+u_ext;
        ode(q_C, t) = v-v_d;

        var v_d: m3_per_s {pub: out};
        var v_d_mm3_s: mm3_per_s {pub: out};

        v_d_mm3_s = v_d/one_mm3;
        v_d       = (u-u_out)/(R/2{dimensionless});
    enddef;

    def comp VP_capillary_H_D as
        //import variables

        var t: second {pub: in};
        var H_global_L: dimensionless {pub: in};
        var H_global_R: dimensionless {pub: in};
        var H_L_out_RHS: dimensionless {pub: in};

        var gamma_mirror: dimensionless {pub: in};
        var tau_H_down: second {pub: in};
        var tau_H_mean: second {pub: in};

        //0.001

        var v_eps: m3_per_s {pub: in};
        var q_C_init: m3 {pub: in};

        //0

        var l: metre {pub: in};
        var r: metre {pub: in};
        var mu_plasma: Js_per_m3 {pub: in};
        var v_in: m3_per_s {pub: in};
        var u_ext: J_per_m3 {pub: in};
        var u_out: J_per_m3 {pub: in};
        var tau_link: second {pub: in};
        var v_scale: m3_per_s {pub: in};
        var one_mm3: m3 {pub: in};
        var w_v: dimensionless {pub: out};

        w_v = 0.5{dimensionless}+1{dimensionless}/pi*atan(v/v_scale);

        ///////////////////////////coupling with RHS vessel using ode to break 
        //instaneous coupling requirement/////////

        var H_link_R: dimensionless {init: H_global_L, pub: out};

        ode(H_link_R, t) = (H_L_out_RHS-H_link_R)/tau_link;

        //RHS means RHS vessel

        var H_link_L: dimensionless {init: H_global_L, pub: out};

        ode(H_link_L, t) = (H_R_out_LHS-H_link_L)/tau_link;

        //RHS means RHS vessel

        var H_L_in: dimensionless {pub: out};

        H_L_in = H_link_L;

        var H_R_in: dimensionless {pub: out};

        H_R_in = H_link_R;

        var H_up: dimensionless {pub: out};

        H_up = w_v*H_L_in+(1{dimensionless}-w_v)*H_R_in;

        var H_down: dimensionless {init: H_global_L, pub: out};
        var H_down_target: dimensionless {pub: out};

        H_down_target  = s_v*(H_mean+gamma_mirror*(H_mean-H_up))+(1{dimensionless}-s_v)*H_mean;
        ode(H_down, t) = (H_down_target-H_down)/tau_H_down;

        var s_v: dimensionless {pub: out};

        s_v = abs(v)/(abs(v)+v_eps);

        ///useful when flow =0, if 0, make H_down=H_mean

        var H_L_out: dimensionless {pub: out};
        var H_R_out: dimensionless {pub: out};

        H_L_out = (1{dimensionless}-w_v)*H_down+w_v*H_L_in;
        H_R_out = w_v*H_down+(1{dimensionless}-w_v)*H_R_in;

        var RBC_volume: m3 {init: RBC_volume_init, pub: out};

        ode(RBC_volume, t) = v_in*H_mass_L-v*H_mass_R;

        var RBC_volume_init: m3 {pub: out};

        RBC_volume_init = H_global_L*q_us;

        //pos/neg for RBC mass conservation only

        var v_pos: m3_per_s {pub: out};
        var v_neg: m3_per_s {pub: out};

        v_pos = w_v*v;

        // inflow from left (v>0)

        v_neg = (1{dimensionless}-w_v)*-v;

        // outflow to left (v<0)

        //var H_target: dimensionless {pub: out};

        //H_target = RBC_volume/q;

        H_mean = RBC_volume/q;

        var H_mean: dimensionless {pub: out};

        //ode(H_mean, t) = (H_target-H_mean)/tau_H_mean;

        var H_mass_L: dimensionless {pub: out};
        var H_mass_R: dimensionless {pub: out};

        H_mass_L = w_v*H_L_in+(1{dimensionless}-w_v)*H_L_out;
        H_mass_R = w_v*H_R_out+(1{dimensionless}-w_v)*H_R_in;

        ////////////////

        var q_C: m3 {init: q_C_init, pub: out};
        var q: m3 {pub: out};
        var q_us: m3 {pub: out};

        q_us = pi*sqr(r)*l;
        q    = q_us+q_C;

        var C: m6_per_J {pub: out};

        C = pi*sqr(8.5e-9{metre})*l/133.322{J_per_m3};

        var Z: dimensionless {pub: out};

        Z = (0.8{dimensionless}+exp(-0.075{dimensionless}*2{dimensionless}*r*1e6{dimensionless}))*(-1{dimensionless}+1{dimensionless}/(1{dimensionless}+pow(10{dimensionless}, -11{dimensionless})*pow(2{dimensionless}*r*1e6{dimensionless}, 12{dimensionless})))+1{dimensionless}/(1{dimensionless}+pow(10{dimensionless}, -11{dimensionless})*pow(2{dimensionless}*r*1e6{dimensionless}, 12{dimensionless}));

        var mu_45: dimensionless;

        mu_45 = 6{dimensionless}*exp(-0.085{per_m}*2{dimensionless}*r*1e6{dimensionless})+3.2{dimensionless}-2.44{dimensionless}*exp(-0.06{dimensionless}*pow(2{per_m}*r*1e6{dimensionless}, 0.645{dimensionless}));

        var mu: Js_per_m3 {pub: out};

        mu = hem_dep_u_rel*mu_plasma;

        var hem_dep_u_rel: dimensionless;

        hem_dep_u_rel = 1{dimensionless}+(mu_45-1{dimensionless})*(pow(1{dimensionless}-H_mean, Z)-1{dimensionless})/(pow(1{dimensionless}-H_global_L, Z)-1{dimensionless})*sqr(2{dimensionless}*r*1e6{dimensionless}/(2{dimensionless}*r*1e6{dimensionless}-1.1{metre}));

        var R: Js_per_m6 {pub: out};

        R = 8{dimensionless}*mu*l/(pi*pow(r, 4{dimensionless}));

        var v: m3_per_s {pub: out};
        var v_mm3_s: mm3_per_s {pub: out};

        v_mm3_s = v/one_mm3;
        v       = (u-u_out)/R;

        var u: J_per_m3 {pub: out};
        var u_mmHg: dimensionless {pub: out};

        u_mmHg      = u/133.322{J_per_m3};
        u           = q_C/C+u_ext;
        ode(q_C, t) = v_in-v;
    enddef;

                H_to2;

            case bc3_is_out == 1{dimensionless}:
                H_daughter3;

        var H_from2: dimensionless {init: H_global_L, pub: out};

            case (is_split == 1{dimensionless}) and (bc3_is_out == 1{dimensionless}) and (alpha3 == 0{dimensionless}) and (Qout3_rem > Qout1_rem) and (Qout3_rem > Qout2_rem) and (Qout3_rem >= Qout4_rem):

        endsel;

        // ---------- D_alpha from which ALPHA port? ----------

        var D_alpha: metre {pub: out};

        D_alpha = sel
            case alpha1 == 1{dimensionless}:
                D1;
            case alpha2 == 1{dimensionless}:
                D2;
            case alpha3 == 1{dimensionless}:
                D3;
            case alpha4 == 1{dimensionless}:
                D4;
            case true:
                D3;
        endsel;

        // ---------- D_beta from which BETA port? ----------

        var D_beta: metre {pub: out};

        D_beta = sel
            case beta1 == 1{dimensionless}:
                D1;
            case beta2 == 1{dimensionless}:
                D2;
            case beta3 == 1{dimensionless}:
                D3;
            case beta4 == 1{dimensionless}:
                D4;
            case true:
                D4;
        endsel;

        // ---------- daughter outflow magnitudes for alpha/beta ----------

        var v_alpha: m3_per_s {pub: out};
        var v_beta: m3_per_s {pub: out};

        v_alpha = sel
            case alpha1 == 1{dimensionless}:
                Qout1;
            case alpha2 == 1{dimensionless}:
                Qout2;
            case alpha3 == 1{dimensionless}:
                Qout3;
            case alpha4 == 1{dimensionless}:
                Qout4;
            case true:
                0{m3_per_s};
        endsel;

        v_beta = sel
            case beta1 == 1{dimensionless}:
                Qout1;
            case beta2 == 1{dimensionless}:
                Qout2;
            case beta3 == 1{dimensionless}:
                Qout3;
            case beta4 == 1{dimensionless}:
                Qout4;
            case true:
                0{m3_per_s};
        endsel;

        var FQB_alpha: dimensionless {pub: out};

        FQB_alpha = (v_alpha+div_0)/(v_alpha+v_beta+2{dimensionless}*div_0);

        var FQE_alpha: dimensionless {pub: out};

        FQE_alpha = 1{dimensionless}/(1{dimensionless}+exp(-(A+B*ph)));

        var B: dimensionless {pub: out};

        B = 1{dimensionless}+6.98{dimensionless}*(1{dimensionless}-H_mean)/(D_F*1e6{dimensionless});

        var A: dimensionless {pub: out};

        A = -6.96{dimensionless}*ln(D_alpha*1e6{dimensionless}/(D_beta*1e6{dimensionless}))/(D_F*1e6{dimensionless});

        var X_0: dimensionless {pub: out};

        X_0 = 0.4{dimensionless}/(D_F*1e6{dimensionless});

        var ph: dimensionless {pub: out};

        ph = ln(y/(1{dimensionless}-y));

        var y_raw: dimensionless {pub: out};

        y_raw = (FQB_alpha-X_0)/(1{dimensionless}-2{dimensionless}*X_0+div_0);

        var y: dimensionless {pub: out};

        y = min(max(y_raw, div_0y), 1{dimensionless}-div_0y);

        var epsilon_alpha: dimensionless {pub: out};

        epsilon_alpha = H_VV_out_alpha-H_mean*FQE_alpha/(FQB_alpha+div_0);

        var H_VV_out_alpha: dimensionless {init: H_global_L, pub: out};

        ode(H_VV_out_alpha, t) = is_split*-k*epsilon_alpha;

        var H_VV_out_beta: dimensionless {pub: out};

        H_VV_out_beta = H_mean*(1{dimensionless}-FQE_alpha)/(1{dimensionless}-FQB_alpha+div_0);

        //////////////////////////////////VV_BG////////////////////////////////
        //////////////////////////////////////////////

        var RBC_in: m3_per_s {pub: out};

        RBC_in = Qin1*H_to1+Qin2*H_to2+Qin3*H_to3+Qin4*H_to4;

        var RBC_out: m3_per_s {pub: out};

        RBC_out = Qout1*H_from1+Qout2*H_from2+Qout3*H_from3+Qout4*H_from4;

        // --- stored RBC volume inside VV1 ---

        var RBC_volume_init: m3 {pub: out};
        var RBC_volume: m3 {init: RBC_volume_init, pub: out};

        RBC_volume_init = H_global_L*q_us;

        // mass balance

        ode(RBC_volume, t) = RBC_in-RBC_out;

        // --- mean VV1 hematocrit (concentration) ---

        var H_mean: dimensionless {pub: out};

        H_mean = RBC_volume/(q_us+div_0);

        ///////////////////////////////setting junctional compliance to greatest compliance of the vessels connected to it 

        var C_conn1: m6_per_J {pub: in};
        var C_conn2: m6_per_J {pub: in};
        var C_conn3: m6_per_J {pub: in};
        var C_conn4: m6_per_J {pub: in};
        var C_max12: m6_per_J {pub: out};
        var C_max123: m6_per_J {pub: out};
        var C: m6_per_J {pub: out};

        C_max12 = sel
            case C_conn1 > C_conn2:
                C_conn1;
            case C_conn1 <= C_conn2:
                C_conn2;
        endsel;

        C_max123 = sel
            case C_max12 > C_conn3:
                C_max12;
            case C_max12 <= C_conn3:
                C_conn3;
        endsel;

        C = sel
            case C_max123 > C_conn4:
                C_max123;
            case C_max123 <= C_conn4:
                C_conn4;
        endsel;

        var q_C: m3 {init: q_C_init, pub: out};
        var q_C_d: m3 {init: q_C_init, pub: out};
        var q: m3 {pub: out};
        var q_us: m3 {pub: out};

        q_us = pi*sqr(r)*l;
        q    = q_us+q_C+q_C_d;

        var R: Js_per_m6 {pub: in};

        // State Variables

        var u: J_per_m3 {pub: out};
        var u_mmHg: dimensionless {pub: out};
        var v: m3_per_s {pub: out};
        var v_mm3_s: mm3_per_s {pub: out};

        v_mm3_s = v/one_mm3;

        var u_d: J_per_m3 {pub: out};
        var u_d_mmHg: dimensionless {pub: out};

        // Conservation Laws
        //question?????????? whether ode (q_c and q_C_d) should be vbc1-4 or vj1-4

        u             = q_C/(C/2{dimensionless})+u_ext;
        ode(q_C, t)   = vbc1+vbc2-v;
        u_mmHg        = u/133.322{J_per_m3};
        v             = (u-u_d)/R;
        u_d           = q_C_d/(C/2{dimensionless})+u_ext;
        ode(q_C_d, t) = v-vbc3-vbc4;
        u_d_mmHg      = u_d/133.322{J_per_m3};
    enddef;
enddef;
